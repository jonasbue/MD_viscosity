\pgfplotsset{compat=newest}
\usepgfplotslibrary{external}
\tikzexternalize[prefix=out/]

\pgfplotsset{
    discard if/.style 2 args={          % This defines pgfplots commands to
        x filter/.code={                % select specific rows from data files.   
            \edef\tempa{\thisrow{#1}}   % With this, all viscosity measurements 
            \edef\tempb{#2}             % can be written to the same file, and 
            \ifx\tempa\tempb            % the data can be selected when
                \def\pgfmathresult{nan} % plotting the figure.       
            \fi
        }
    },
    discard if not/.style 2 args={      % Name of command
        x filter/.code={                % Filter
            \edef\tempa{\thisrow{#1}}   % Define temporary variable: row.
            \edef\tempb{#2}             % Temp variable 2
            \ifx\tempa\tempb            % If tempa == tempb, do nothing
            \else                       % else,
                \def\pgfmathresult{nan} % return not a number.
            \fi
        }
    }
}
